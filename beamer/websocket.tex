\begin{frame}
\frametitle{WebSocket Protocol}
\begin{itemize}
\item geringer Overhead (zwei Byte)
\item Begin: 0x00
\item Ende: OxFF
\item Daten in UTF-8
\item Beispiel: \textbackslash 0x00Hello, WebSocket\textbackslash 0xff
\end{itemize}
\end{frame}
